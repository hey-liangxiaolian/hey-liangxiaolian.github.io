深拷贝之递归拷贝

const obj = {
    name: 'anan',
    data: [1, 2],
    b: {
        c: 'ssss',
    },
    eat() {
        console.log('aaa')
    },
}

/* 
    函数说明：实现对象的深拷贝
    params：
        obj：待拷贝的对象
    return：
        newObj：新拷贝的对象
*/

function deepClone(obj) {
    // 首先判断是否是数组，如果是数组就将赋值的内容放入[]中，否则放入{}
    let newObj = obj instanceof Array ? [] : {}
    for (const k in obj) { // for in会遍历原型链上的属性
        if (obj.hasOwnProperty(k)) {  //所以需要用hasOwnProperty判断是否有自己的属性
            newObj[k] = typeof obj[k] === 'object' ? deepClone(obj[k]) : obj[k]
        }
    }
    return newObj
}

或使用for of遍历

function deepClone(obj) {
    // 首先判断是否是数组，如果是数组就将赋值的内容放入[]中，否则放入{}
    let newObj = obj instanceof Array ? [] : {}
    // Object.entries()可以将对象转换为可迭代的对象 这样就可以使用 for / of 来遍历
    for (const [k, v] of Object.entries(obj)) {
        // 递归实现
        // 再判断是对象还是普通的数据类型
        // 如果是对象就通过递归来进一步判断
        // 如果是普通数据类型就直接复制就好
        newObj[k] = typeof v == 'object' ? deepClone(v) : v
    }
    return newObj
}

调用函数：

let obj2 = deepClone(obj)
obj2.data.push('cuc')
const obj3 = JSON.parse(JSON.stringify(obj))
console.log(obj)
console.log(obj2)
console.log(obj3)
